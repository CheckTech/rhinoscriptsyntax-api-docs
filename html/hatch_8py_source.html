24 def AddHatches(curve_ids, hatch_pattern=None, scale=1.0, rotation=0.0):
25  """Creates one or more new hatch objects a list of closed planar curves
26  Parameters:
27  curve_ids = identifiers of the closed planar curves that defines the
28  boundary of the hatch objects
29  hatch_pattern[opt] = name of the hatch pattern to be used by the hatch
30  object. If omitted, the current hatch pattern will be used
31  scale[opt] = hatch pattern scale factor
32  rotation[opt] = hatch pattern rotation angle in degrees.
33  Returns:
34  identifiers of the newly created hatch on success
35  None on error
36  """
37  id = rhutil.coerceguid(curve_ids, False)
38  if id: curve_ids = [id]
39  index = scriptcontext.doc.HatchPatterns.CurrentHatchPatternIndex
40  if hatch_pattern and hatch_pattern!=index:
41  if isinstance(hatch_pattern, int):
42  index = hatch_pattern
43  else:
44  index = scriptcontext.doc.HatchPatterns.Find(hatch_pattern, True)
45  if index<0: return scriptcontext.errorhandler()
46  curves = [rhutil.coercecurve(id, -1, True) for id in curve_ids]
47  rotation = Rhino.RhinoMath.ToRadians(rotation)
48  hatches = Rhino.Geometry.Hatch.Create(curves, index, rotation, scale)
49  if not hatches: return scriptcontext.errorhandler()
50  ids = []
51  for hatch in hatches:
52  id = scriptcontext.doc.Objects.AddHatch(hatch)
53  if id==System.Guid.Empty: continue
54  ids.append(id)
55  if not ids: return scriptcontext.errorhandler()
56  scriptcontext.doc.Views.Redraw()
57  return ids

101 def ExplodeHatch(hatch_id, delete=False):
102  """Explodes a hatch object into its component objects. The exploded objects
103  will be added to the document. If the hatch object uses a solid pattern,
104  then planar face Brep objects will be created. Otherwise, line curve objects
105  will be created
106  Parameters:
107  hatch_id = identifier of a hatch object
108  delete[opt] = delete the hatch object
109  Returns:
110  list of identifiers for the newly created objects
111  None on error
112  """
113  rhobj = rhutil.coercerhinoobject(hatch_id, True, True)
114  pieces = rhobj.HatchGeometry.Explode()
115  if not pieces: return scriptcontext.errorhandler()
116  attr = rhobj.Attributes
117  rc = []
118  for piece in pieces:
119  id = None
120  if isinstance(piece, Rhino.Geometry.Curve):
121  id = scriptcontext.doc.Objects.AddCurve(piece, attr)
122  elif isinstance(piece, Rhino.Geometry.Brep):
123  id = scriptcontext.doc.Objects.AddBrep(piece, attr)
124  if id: rc.append(id)
125  if delete: scriptcontext.doc.Objects.Delete(rhobj)
126  return rc

129 def HatchPattern(hatch_id, hatch_pattern=None):
130  """Returns or changes a hatch object's hatch pattern
131  Paramters:
132  hatch_id = identifier of a hatch object
133  hatch_pattern[opt] = name of an existing hatch pattern to replace the
134  current hatch pattern
135  Returns:
136  if hatch_pattern is not specified, the current hatch pattern
137  if hatch_pattern is specified, the previous hatch pattern
138  None on error
139  """
140  hatchobj = rhutil.coercerhinoobject(hatch_id, True, True)
141  if not isinstance(hatchobj, Rhino.DocObjects.HatchObject):
142  return scriptcontext.errorhandler()
143  old_index = hatchobj.HatchGeometry.PatternIndex

146  if new_index<0: return scriptcontext.errorhandler()
147  hatchobj.HatchGeometry.PatternIndex = new_index
148  hatchobj.CommitChanges()

150  return scriptcontext.doc.HatchPatterns[old_index].Name
