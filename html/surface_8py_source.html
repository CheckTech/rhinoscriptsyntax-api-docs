161 def AddNurbsSurface(point_count, points, knots_u, knots_v, degree, weights=None):
162  """Adds a NURBS surface object to the document
163  Parameters:
164  point_count = number of control points in the u and v direction
165  points = list of 3D points
166  knots_u = knot values for the surface in the u direction.
167  Must contain point_count[0]+degree[0]-1 elements
168  knots_v = knot values for the surface in the v direction.
169  Must contain point_count[1]+degree[1]-1 elements
170  degree = degree of the surface in the u and v directions.
171  weights[opt] = weight values for the surface. The number of elements in
172  weights must equal the number of elements in points. Values must be
173  greater than zero.
174  Returns:
175  identifier of new object if successful
176  None on error
177  """
178  if len(points)<(point_count[0]*point_count[1]):
179  return scriptcontext.errorhandler()
180  ns = Rhino.Geometry.NurbsSurface.Create(3, weights!=None, degree[0]+1, degree[1]+1, point_count[0], point_count[1])
181  #add the points and weights
182  controlpoints = ns.Points
183  index = 0
184  for i in range(point_count[0]):
185  for j in range(point_count[1]):
186  if weights:
187  cp = Rhino.Geometry.ControlPoint(points[index], weights[index])
188  controlpoints.SetControlPoint(i,j,cp)
189  else:
190  cp = Rhino.Geometry.ControlPoint(points[index])
191  controlpoints.SetControlPoint(i,j,cp)
192  index += 1
193  #add the knots
194  for i in range(ns.KnotsU.Count): ns.KnotsU[i] = knots_u[i]
195  for i in range(ns.KnotsV.Count): ns.KnotsV[i] = knots_v[i]
196  if not ns.IsValid: return scriptcontext.errorhandler()
197  id = scriptcontext.doc.Objects.AddSurface(ns)
198  if id==System.Guid.Empty: return scriptcontext.errorhandler()
199  scriptcontext.doc.Views.Redraw()
200  return id

202 def AddPatch(object_ids, uv_spans_tuple_OR_surface_object_id, tolerance=None, trim=True, point_spacing=0.1, flexibility=1.0, surface_pull=1.0, fix_edges=False):
203  """Fits a surface through curve, point, point cloud, and mesh objects.
204  Parameters:
205  object_ids = a list of object identifiers that indicate the objects to use for the patch fitting.
206  Acceptable object types include curves, points, point clouds, and meshes.
207  uv_spans_tuple_OR_surface_object_id = the U and V direction span counts for the automatically generated surface OR
208  The identifier of the starting surface. It is best if you create a starting surface that is similar in shape
209  to the surface you are trying to create.
210  tolerance[opt] = The tolerance used by input analysis functions. If omitted, Rhino's document absolute tolerance is used.
211  trim[opt] = Try to find an outside curve and trims the surface to it. The default value is True.
212  point_spacing[opt] = The basic distance between points sampled from input curves. The default value is 0.1.
213  flexibility[opt] = Determines the behavior of the surface in areas where its not otherwise controlled by the input.
214  Lower numbers make the surface behave more like a stiff material, higher, more like a flexible material.
215  That is, each span is made to more closely match the spans adjacent to it if there is no input geometry
216  mapping to that area of the surface when the flexibility value is low. The scale is logarithmic.
217  For example, numbers around 0.001 or 0.1 make the patch pretty stiff and numbers around 10 or 100
218  make the surface flexible. The default value is 1.0.
219  surface_pull[opt] = Similar to stiffness, but applies to the starting surface. The bigger the pull, the closer
220  the resulting surface shape will be to the starting surface. The default value is 1.0.
221  fix_edges[opt] = Clamps the edges of the starting surface in place. This option is useful if you are using a
222  curve or points for deforming an existing surface, and you do not want the edges of the starting surface
223  to move. The default if False.
224  Returns:
225  Identifier of the new surface object if successful.
226  None on error.
227  """
228  # System.Collections.List instead of Python list because IronPython is
229  # having problems converting a list to an IEnumerable<GeometryBase> which
230  # is the 1st argument for Brep.CreatePatch
231  geometry = List[Rhino.Geometry.GeometryBase]()
232  u_span = 10
233  v_span = 10
234  rc = None
235  id = rhutil.coerceguid(object_ids, False)
236  if id: object_ids = [id]
237  for object_id in object_ids:
238  rhobj = rhutil.coercerhinoobject(object_id, False, False)
239  if not rhobj: return None
240  geometry.Add( rhobj.Geometry )
241  if not geometry: return None
242 
243  surface = None
244  if uv_spans_tuple_OR_surface_object_id:
245  if type(uv_spans_tuple_OR_surface_object_id) is tuple:
246  u_span = uv_spans_tuple_OR_surface_object_id[0]
247  v_span = uv_spans_tuple_OR_surface_object_id[1]
248  else:
249  surface = rhutil.coercesurface(uv_spans_tuple_OR_surface_object_id, False)
250 
251  if not tolerance: tolerance = scriptcontext.doc.ModelAbsoluteTolerance
252  b = System.Array.CreateInstance(bool, 4)
253  for i in range(4): b[i] = fix_edges
254  brep = Rhino.Geometry.Brep.CreatePatch(geometry, surface, u_span, v_span, trim, False, point_spacing, flexibility, surface_pull, b, tolerance)
255  if brep:
256  rc = scriptcontext.doc.Objects.AddBrep(brep)
257  scriptcontext.doc.Views.Redraw()
258  return rc

325 def AddLoftSrf(object_ids, start=None, end=None, loft_type=0, simplify_method=0, value=0, closed=False):
326  """Adds a surface created by lofting curves to the document.
327  - no curve sorting performed. pass in curves in the order you want them sorted
328  - directions of open curves not adjusted. Use CurveDirectionsMatch and
329  ReverseCurve to adjust the directions of open curves
330  - seams of closed curves are not adjusted. Use CurveSeam to adjust the seam
331  of closed curves
332  Parameters:
333  object_ids = ordered list of the curves to loft through
334  start [opt] = starting point of the loft
335  end [opt] = ending point of the loft
336  loft_type [opt] = type of loft. Possible options are:
337  0 = Normal. Uses chord-length parameterization in the loft direction
338  1 = Loose. The surface is allowed to move away from the original curves
339  to make a smoother surface. The surface control points are created
340  at the same locations as the control points of the loft input curves.
341  2 = Straight. The sections between the curves are straight. This is
342  also known as a ruled surface.
343  3 = Tight. The surface sticks closely to the original curves. Uses square
344  root of chord-length parameterization in the loft direction
345  4 = Developable. Creates a separate developable surface or polysurface
346  from each pair of curves.
347  simplify_method [opt] = Possible options are:
348  0 = None. Does not simplify.
349  1 = Rebuild. Rebuilds the shape curves before lofting.
350  2 = Refit. Refits the shape curves to a specified tolerance
351  value [opt] = A value based on the specified style.
352  style=1 (Rebuild), then value is the number of control point used to rebuild
353  style=1 is specified and this argument is omitted, then curves will be
354  rebuilt using 10 control points.
355  style=2 (Refit), then value is the tolerance used to rebuild.
356  style=2 is specified and this argument is omitted, then the document's
357  absolute tolerance us used for refitting.
358  Returns:
359  Array containing the identifiers of the new surface objects if successful
360  None on error
361  """
362  if loft_type<0 or loft_type>5: raise ValueError("loft_type must be 0-4")
363  if simplify_method<0 or simplify_method>2: raise ValueError("simplify_method must be 0-2")
364 
365  # get set of curves from object_ids
366  curves = [rhutil.coercecurve(id,-1,True) for id in object_ids]
367  if len(curves)<2: return scriptcontext.errorhandler()
368  if start is None: start = Rhino.Geometry.Point3d.Unset
369  if end is None: end = Rhino.Geometry.Point3d.Unset
370  start = rhutil.coerce3dpoint(start, True)
371  end = rhutil.coerce3dpoint(end, True)
372 
373  lt = Rhino.Geometry.LoftType.Normal
374  if loft_type==1: lt = Rhino.Geometry.LoftType.Loose
375  elif loft_type==2: lt = Rhino.Geometry.LoftType.Straight
376  elif loft_type==3: lt = Rhino.Geometry.LoftType.Tight
377  elif loft_type==4: lt = Rhino.Geometry.LoftType.Developable
378 
379  breps = None
380  if simplify_method==0:
381  breps = Rhino.Geometry.Brep.CreateFromLoft(curves, start, end, lt, closed)
382  elif simplify_method==1:
383  value = abs(value)
384  rebuild_count = int(value)
385  breps = Rhino.Geometry.Brep.CreateFromLoftRebuild(curves, start, end, lt, closed, rebuild_count)
386  elif simplify_method==2:
387  refit = abs(value)
388  if refit==0: refit = scriptcontext.doc.ModelAbsoluteTolerance
389  breps = Rhino.Geometry.Brep.CreateFromLoftRefit(curves, start, end, lt, closed, refit)
390  if not breps: return scriptcontext.errorhandler()
391 
392  idlist = []
393  for brep in breps:
394  id = scriptcontext.doc.Objects.AddBrep(brep)
395  if id!=System.Guid.Empty: idlist.append(id)
396  if idlist: scriptcontext.doc.Views.Redraw()
397  return idlist
