669 def ObjectsByType(geometry_type, select=False, state=0):
670  """Returns identifiers of all objects based on the objects' geometry type.
671  Parameters:
672  geometry_type = The type(s) of geometry objects (points, curves, surfaces,
673  meshes, etc.) that can be selected. Object types can be
674  added together to filter several different kinds of geometry.
675  Value Description
676  0 All objects
677  1 Point
678  2 Point cloud
679  4 Curve
680  8 Surface or single-face brep
681  16 Polysurface or multiple-face
682  32 Mesh
683  256 Light
684  512 Annotation
685  4096 Instance or block reference
686  8192 Text dot object
687  16384 Grip object
688  32768 Detail
689  65536 Hatch
690  131072 Morph control
691  134217728 Cage
692  268435456 Phantom
693  536870912 Clipping plane
694  select[opt] = Select the objects
695  state[opt] = Object state. See help
696  Returns:
697  A list of Guids identifying the objects.
698  """
699  if not state: state = 7
700  bSurface = False
701  bPolySurface = False
702  bLights = False
703  bGrips = False
704  bPhantoms = False
705  geometry_filter = __FilterHelper(geometry_type)
706  if type(geometry_type) is int and geometry_type==0:
707  geometry_filter = Rhino.DocObjects.ObjectType.AnyObject
708  if geometry_filter & Rhino.DocObjects.ObjectType.Surface: bSurface = True
709  if geometry_filter & Rhino.DocObjects.ObjectType.Brep: bPolySurface = True
710  if geometry_filter & Rhino.DocObjects.ObjectType.Light: bLights = True
711  if geometry_filter & Rhino.DocObjects.ObjectType.Grip: bGrips = True
712  if geometry_filter & Rhino.DocObjects.ObjectType.Phantom: bPhantoms = True
713 
714  it = Rhino.DocObjects.ObjectEnumeratorSettings()
715  it.DeletedObjects = False
716  it.ActiveObjects = True
717  it.ReferenceObjects = True
718  it.IncludeLights = bLights
719  it.IncludeGrips = bGrips
720  it.IncludePhantoms = bPhantoms
721 
722  it.NormalObjects = True
723  it.LockedObjects = True
724  it.HiddenObjects = True
725 
726  object_ids = []
727  e = scriptcontext.doc.Objects.GetObjectList(it)
728  for object in e:
729  if state & 1 and object.IsNormal or state & 2 and object.IsLocked or state & 4 and object.IsHidden:
730  bFound = False
731  object_type = object.ObjectType
732  if object_type==Rhino.DocObjects.ObjectType.Brep and (bSurface or bPolySurface):
733  brep = rhutil.coercebrep(object.Id)
734  if brep:
735  if brep.Faces.Count==1:
736  if bSurface: bFound = True
737  else:
738  if bPolySurface: bFound = True
739  elif object_type==Rhino.DocObjects.ObjectType.Extrusion and (bSurface or bPolySurface):
740  extrusion = object.Geometry
741  profile_count = extrusion.ProfileCount
742  cap_count = extrusion.CapCount
743  if profile_count==1 and cap_count==0 and bSurface:
744  bFound = True
745  elif profile_count>0 and cap_count>0 and bPolySurface:
746  bFound = True
747  elif object_type & geometry_filter:
748  bFound = True
749 
750  if bFound:
751  if select: object.Select(True)
752  object_ids.append(object.Id)
753 
754  if object_ids and select: scriptcontext.doc.Views.Redraw()
755  return object_ids

812 def WindowPick(corner1, corner2, view=None, select=False, in_window=True):
813  """Picks objects using either a window or crossing selection
814  Parameters:
815  corner1, corner2 = corners of selection window
816  view[opt] = view to perform the selection in
817  select[opt] = select picked objects
818  in_window[opt] = if False, then a crossing window selection is performed
819  Returns:
820  list of object ids on success
821  """
822  viewport = __viewhelper(view).MainViewport
823  screen1 = Rhino.Geometry.Point2d(rhutil.coerce3dpoint(corner1, True))
824  screen2 = Rhino.Geometry.Point2d(rhutil.coerce3dpoint(corner2, True))
825  xf = viewport.GetTransform(Rhino.DocObjects.CoordinateSystem.World, Rhino.DocObjects.CoordinateSystem.Screen)
826  screen1.Transform(xf)
827  screen2.Transform(xf)
828  objects = None
829  filter = Rhino.DocObjects.ObjectType.AnyObject
830  if in_window:
831  objects = scriptcontext.doc.Objects.FindByWindowRegion(viewport, screen1, screen2, True, filter)
832  else:
833  objects = scriptcontext.doc.Objects.FindByCrossingWindowRegion(viewport, screen1, screen2, True, filter)
834  if objects:
835  rc = []
836  for rhobj in objects:
837  rc.append(rhobj.Id)
838  if select: rhobj.Select(True)
839  if select: scriptcontext.doc.Views.Redraw()
840  return rc
