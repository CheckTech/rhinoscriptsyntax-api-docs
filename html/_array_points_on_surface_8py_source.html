4 def ArrayPointsOnSurface():
5  # Get the surface object
6  surface_id = rs.GetObject("Select surface", rs.filter.surface)
7  if surface_id is None: return
8 
9  # Get the number of rows
10  rows = rs.GetInteger("Number of rows", 2, 2)
11  if rows is None: return
12 
13  # Get the number of columns
14  columns = rs.GetInteger("Number of columns", 2, 2)
15  if columns is None: return
16 
17  # Get the domain of the surface
18  U = rs.SurfaceDomain(surface_id, 0)
19  V = rs.SurfaceDomain(surface_id, 1)
20  if U is None or V is None: return
21 
22  # Add the points
23  for i in xrange(0,rows):
24  param0 = U[0] + (((U[1] - U[0]) / (rows-1)) * i)
25  for j in xrange(0,columns):
26  param1 = V[0] + (((V[1] - V[0]) / (columns-1)) * j)
27  point = rs.EvaluateSurface(surface_id, param0, param1)
28  rs.AddPoint(point)
