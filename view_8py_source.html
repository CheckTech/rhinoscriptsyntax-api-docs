21  Parameters:

23  corner1, corner2 = 2d corners of the detail in the layout's unit system
24  title[opt] = title of the new detail

26  1 = parallel top view

454 def RotateCamera(view=None, direction=0, angle=None):
455  """Rotates a perspective-projection view's camera. See the RotateCamera
456  command in the Rhino help file for more details
457  Parameters:
458  view:[opt] title or id of the view. If omitted, current active view is used
459  direction: [opt] the direction to rotate the camera where 0=right, 1=left,
460  2=down, 3=up
461  angle: [opt] the angle to rotate. If omitted, the angle of rotation
462  is specified by the "Increment in divisions of a circle" parameter
463  specified in Options command's View tab
464  Returns:
465  True or False indicating success or failure
466  """
467  view = __viewhelper(view)
468  viewport = view.ActiveViewport
469  if angle is None:
470  angle = 2.0*math.pi/Rhino.ApplicationSettings.ViewSettings.RotateCircleIncrement
471  else:
472  angle = Rhino.RhinoMath.ToRadians( abs(angle) )
473  target_distance = (viewport.CameraLocation-viewport.CameraTarget)*viewport.CameraZ
474  axis = viewport.CameraY
475  if direction==0 or direction==2: angle=-angle
476  if direction==0 or direction==1:
477  if Rhino.ApplicationSettings.ViewSettings.RotateToView:
478  axis = viewport.CameraY
479  else:
480  axis = Rhino.Geometry.Vector3d.ZAxis
481  elif direction==2 or direction==3:
482  axis = viewport.CameraX
483  else:
484  return False
485  if Rhino.ApplicationSettings.ViewSettings.RotateReverseKeyboard: angle=-angle
486  rot = Rhino.Geometry.Transform.Rotation(angle, axis, Rhino.Geometry.Point3d.Origin)
487  camUp = rot * viewport.CameraY
488  camDir = -(rot * viewport.CameraZ)
489  target = viewport.CameraLocation + target_distance*camDir
490  viewport.SetCameraLocations(target, viewport.CameraLocation)
491  viewport.CameraUp = camUp
492  view.Redraw()
493  return True

833 def ViewRadius(view=None, radius=None, mode=False):
834  """Returns or sets the radius of a parallel-projected view. Useful
835  when you need an absolute zoom factor for a parallel-projected view
836  Parameters:
837  view:[opt] title or id of the view. If omitted, current active view is used
838  radius:[opt] the view radius
839  mode: [opt] perform a "dolly" magnification by moving the camera
840  towards/away from the target so that the amount of the screen
841  subtended by an object changes. true = perform a "zoom"
842  magnification by adjusting the "lens" angle
843  Returns:
844  if radius is not specified, the current view radius for the specified view
845  if radius is specified, the previous view radius for the specified view
846  """
847  view = __viewhelper(view)
848  viewport = view.ActiveViewport
849  if not viewport.IsParallelProjection: return scriptcontext.errorhandler()
850  fr = viewport.GetFrustum()
851  frus_right = fr[2]
852  frus_top = fr[4]
853  old_radius = min(frus_top, frus_right)
854  if radius is None: return old_radius
855  magnification_factor = radius / old_radius
856  d = 1.0 / magnification_factor
857  viewport.Magnify(d, mode)
858  view.Redraw()
859  return old_radius
