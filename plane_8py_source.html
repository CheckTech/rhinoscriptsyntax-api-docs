77 def PlaneCurveIntersection(plane, curve, tolerance=None):
78  "Intersect an infinite plane and a curve object"
79  plane = rhutil.coerceplane(plane, True)
80  curve = rhutil.coercecurve(curve, -1, True)
81  if tolerance is None: tolerance = scriptcontext.doc.ModelAbsoluteTolerance
82  intersections = Rhino.Geometry.Intersect.Intersection.CurvePlane(curve, plane, tolerance)
83  if intersections:
84  rc = []
85  for intersection in intersections:
86  a = 1
87  if intersection.IsOverlap: a = 2
88  b = intersection.PointA
89  c = intersection.PointA2
90  d = intersection.PointB
91  e = intersection.PointB2
92  f = intersection.ParameterA
93  g = intersection.ParameterB
94  h = intersection.OverlapA[0]
95  i = intersection.OverlapA[1]
96  j = intersection.OverlapB[0]
97  k = intersection.OverlapB[1]
98  rc.append( (a,b,c,d,e,f,g,h,i,j,k) )
99  return rc

143  """Creates a plane from an origin point and a normal direction vector.
144  Parameters:
145  origin = A 3D point identifying the origin of the plane.

190 def PlaneSphereIntersection(plane, sphere_plane, sphere_radius):
191  """Calculates the intersection of a plane and a sphere
192  Parameters:
193  plane = the plane to intersect
194  sphere_plane = equitorial plane of the sphere. origin of the plane is
195  the center of the sphere
196  sphere_radius = radius of the sphere
197  Returns:
198  list of intersection results - see help
199  None on error
200  """
201  plane = rhutil.coerceplane(plane, True)
202  sphere_plane = rhutil.coerceplane(sphere_plane, True)
203  sphere = Rhino.Geometry.Sphere(sphere_plane, sphere_radius)
204  rc, circle = Rhino.Geometry.Intersect.Intersection.PlaneSphere(plane, sphere)
205  if rc==Rhino.Geometry.Intersect.PlaneSphereIntersection.Point:
206  return 0, circle.Center
207  if rc==Rhino.Geometry.Intersect.PlaneSphereIntersection.Circle:
208  return 1, circle.Plane, circle.Radius
