224 def XformScale(scale, point=None):
225  """Creates a scale transformation
226  Parameters:
227  scale = single number, list of 3 numbers, Point3d, or Vector3d
228  point[opt] = center of scale. If omitted, world origin is used
229  Returns:
230  The 4x4 transformation matrix on success
231  None on error

233  factor = rhutil.coerce3dpoint(scale)

245 def XformScreenToWorld(point, view=None, screen_coordinates=False):
246  """Transforms a point from either client-area coordinates of the specified view
247  or screen coordinates to world coordinates. The resulting coordinates are represented
248  as a 3-D point
249  Parameters:
250  point = 2D point
251  view[opt] = title or identifier of a view. If omitted, the active view is used
252  screen_coordinates[opt] = if False, point is in client-area coordinates. If True,
253  point is in screen-area coordinates
254  Returns:
255  3D point on success
256  None on error
257  """
258  point = rhutil.coerce2dpoint(point, True)
259  view = rhview.__viewhelper(view)
260  viewport = view.MainViewport
261  xform = viewport.GetTransform(Rhino.DocObjects.CoordinateSystem.Screen, Rhino.DocObjects.CoordinateSystem.World)
262  point3d = Rhino.Geometry.Point3d(point.X, point.Y, 0)
263  if screen_coordinates:
264  screen = view.ScreenRectangle
265  point3d.X = point.X - screen.Left
266  point3d.Y = point.Y - screen.Top
267  point3d = xform * point3d
268  return point3d

307  """Transforms a point from world coordinates to either client-area coordinates of
