7 def CreatePreviewImage(filename, view=None, size=None, flags=0, wireframe=False):
8  """Create a bitmap preview image of the current model
9  Parameters:
10  filename = name of the bitmap file to create
11  view[opt] = title of the view. If omitted, the active view is used
12  size[opt] = two integers that specify width and height of the bitmap
13  flags[opt] = Bitmap creation flags. Can be the combination of:
14  1 = honor object highlighting
15  2 = draw construction plane
16  4 = use ghosted shading
17  wireframe[opt] = If True then a wireframe preview image. If False,
18  a rendered image will be created
19  Returns:
20  True or False indicating success or failure
21  """
22  rhview = scriptcontext.doc.Views.ActiveView
23  if view is not None:
24  rhview = scriptcontext.doc.Views.Find(view, False)
25  if rhview is None: return False
26  rhsize = rhview.ClientRectangle.Size
27  if size: rhsize = System.Drawing.Size(size[0], size[1])
28  ignore_highlights = (flags&1)!=1
29  drawcplane = (flags&2)==2
30  useghostedshading = (flags&4)==4
31  if wireframe:
32  return rhview.CreateWireframePreviewImage(filename, rhsize, ignore_highlights, drawcplane)
33  else:
34  return rhview.CreateShadedPreviewImage(filename, rhsize, ignore_highlights, drawcplane, useghostedshading)

182 def RenderSettings(settings=None):
183  """Returns or sets render settings
184  Parameters:
185  settings[opt] = render settings to modify.
186  0=none,
187  1=create shadows,
188  2=use lights on layers that are off,
189  4=render curves and isocurves,
190  8=render dimensions and text
191  Returns:
192  if settings are not specified, the current render settings
193  if settings are specified, the previous render settings
194  """
195  rc = 0
196  rendersettings = scriptcontext.doc.RenderSettings
197  if rendersettings.ShadowmapLevel: rc+=1
198  if rendersettings.UseHiddenLights: rc+=2
199  if rendersettings.RenderCurves: rc+=4
200  if rendersettings.RenderAnnotations: rc+=8
201  if settings is not None:
202  rendersettings.ShadowmapLevel = (settings & 1)
203  rendersettings.UseHiddenLights = (settings & 2)==2
204  rendersettings.RenderCurves = (settings & 4)==4
205  rendersettings.RenderAnnotations = (settings & 8)==8
206  scriptcontext.doc.RenderSettings = rendersettings
207  return rc

334 def UnitSystem(unit_system=None, scale=False, in_model_units=True):
335  """Return or set the document's unit system. See Rhino's DocumentProperties
336  command (Units and Page Units Window) for details
337  Parameters:
338  unit_system = The unit system to set the document to. The unit systems are:
339  0 - No unit system
340  1 - Microns (1.0e-6 meters)
341  2 - Millimeters (1.0e-3 meters)
342  3 - Centimeters (1.0e-2 meters)
343  4 - Meters
344  5 - Kilometers (1.0e+3 meters)
345  6 - Microinches (2.54e-8 meters, 1.0e-6 inches)
346  7 - Mils (2.54e-5 meters, 0.001 inches)
347  8 - Inches (0.0254 meters)
348  9 - Feet (0.3408 meters, 12 inches)
349  10 - Miles (1609.344 meters, 5280 feet)
350  11 - *Reserved for custom Unit System*
351  12 - Angstroms (1.0e-10 meters)
352  13 - Nanometers (1.0e-9 meters)
353  14 - Decimeters (1.0e-1 meters)
354  15 - Dekameters (1.0e+1 meters)
355  16 - Hectometers (1.0e+2 meters)
356  17 - Megameters (1.0e+6 meters)
357  18 - Gigameters (1.0e+9 meters)
358  19 - Yards (0.9144 meters, 36 inches)
359  20 - Printer point (1/72 inches, computer points)
360  21 - Printer pica (1/6 inches, (computer picas)
361  22 - Nautical mile (1852 meters)
362  23 - Astronomical (1.4959787e+11)
363  24 - Lightyears (9.46073e+15 meters)
364  25 - Parsecs (3.08567758e+16)
365  scale [opt] = Scale existing geometry based on the new unit system.
366  If not specified, any existing geometry is not scaled (False)
367  in_model_units [opt] = Return or modify the document's model units (True)
368  or the document's page units (False). The default is True.
369  Returns:
370  if unit_system is not specified, the current unit system
371  if unit_system is specified, the previous unit system
372  None on error
373  """
374  if (unit_system is not None and (unit_system<1 or unit_system>25)):
375  raise ValueError("unit_system value of %s is not valid"%unit_system)
376  if in_model_units:
377  rc = int(scriptcontext.doc.ModelUnitSystem)
378  if unit_system is not None:
379  unit_system = System.Enum.ToObject(Rhino.UnitSystem, unit_system)
380  scriptcontext.doc.AdjustModelUnitSystem(unit_system, scale)
381  else:
382  rc = int(scriptcontext.doc.PageUnitSystem)
383  if unit_system is not None:
384  unit_system = System.Enum.ToObject(Rhino.UnitSystem, unit_system)
385  scriptcontext.doc.AdjustPageUnitSystem(unit_system, scale)
386  return rc
