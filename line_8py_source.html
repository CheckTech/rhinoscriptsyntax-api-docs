13 def LineCylinderIntersection(line, cylinder_plane, cylinder_height, cylinder_radius):
14  """Calculates the intersection of a line and a cylinder
15  Parameters:
16  line = the line to intersect
17  cylinder_plane = base plane of the cylinder
18  cylinder_height = height of the cylinder
19  cylinder_radius = radius of the cylinder
20  Returns:
21  list of intersection points (0, 1, or 2 points)
22  """
23  line = rhutil.coerceline(line, True)
24  cylinder_plane = rhutil.coerceplane(cylinder_plane, True)
25  circle = Rhino.Geometry.Circle( cylinder_plane, cylinder_radius )
26  if not circle.IsValid: raise ValueError("unable to create valid circle with given plane and radius")
27  cyl = Rhino.Geometry.Cylinder( circle, cylinder_height )
28  if not cyl.IsValid: raise ValueError("unable to create valid cylinder with given circle and height")
29  rc, pt1, pt2 = Rhino.Geometry.Intersect.Intersection.LineCylinder(line, cyl)
30  if rc==Rhino.Geometry.Intersect.LineCylinderIntersection.None:
31  return []
32  if rc==Rhino.Geometry.Intersect.LineCylinderIntersection.Single:
33  return [pt1]
34  return [pt1, pt2]
